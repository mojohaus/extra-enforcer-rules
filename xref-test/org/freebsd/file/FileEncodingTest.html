1   package org.freebsd.file;
2   
3   import static org.junit.Assert.assertEquals;
4   
5   import java.nio.charset.StandardCharsets;
6   
7   import org.junit.Before;
8   import org.junit.Test;
9   
10  public class FileEncodingTest
11  {
12  
13      private FileEncoding rule;
14  
15      @Before
16      public void initFields()
17      {
18          rule = new FileEncoding();
19      }
20  
21      @Test
22      public void detectAscii()
23      {
24          assertEncoding( new byte[]{'a', 'b', 'c', 'A', 'B', 'C', '1', '2', '3'}, StandardCharsets.US_ASCII.name(), "ASCII" );
25      }
26  
27      @Test
28      public void detectISO()
29      {
30          assertEncoding( new byte[]{'a', 'b', 'c', 'A', 'B', 'C', '1', '2', '3', (byte) 0xF7}, StandardCharsets.ISO_8859_1.name(), "ISO-8859" );
31      }
32  
33      @Test
34      public void detectUTF7()
35      {
36          assertEncoding( new byte[]{'+', '/', 'v', '8', 'B', 'C', '1', '2', '3'}, "UTF-7", "UTF-7 UNICODE" );
37          assertEncoding( new byte[]{'+', '/', 'v', '9', 'B', 'C', '1', '2', '3'}, "UTF-7", "UTF-7 UNICODE" );
38          assertEncoding( new byte[]{'+', '/', 'v', '+', 'B', 'C', '1', '2', '3'}, "UTF-7", "UTF-7 UNICODE" );
39          assertEncoding( new byte[]{'+', '/', 'v', '/', 'B', 'C', '1', '2', '3'}, "UTF-7", "UTF-7 UNICODE" );
40  
41      }
42  
43      @Test
44      public void detectUTF_8()
45      {
46          assertEncoding( new byte[]{'a', 'b', 'c', 'A', 'B', 'C', '1', '2', '3', (byte) 0xC3, (byte) 0xB6}, StandardCharsets.UTF_8.name(), "UTF-8 UNICODE" );
47      }
48  
49      @Test
50      public void detectUTF_8WithBoom()
51      {
52          assertEncoding( new byte[]{(byte) 0xEF, (byte) 0xBB, (byte) 0xBF, 'a', 'b', 'c', 'A', 'B', 'C', '1', '2', '3'}, StandardCharsets.UTF_8.name(), "UTF-8 UNICODE (WITH BOM)" );
53          assertEncoding( new byte[]{(byte) 0xEF, (byte) 0xBB, (byte) 0xBF, 'a', 'b', 'c', 'A', 'B', 'C', '1', '2', '3', (byte) 0xC3, (byte) 0xB6}, StandardCharsets.UTF_8.name(), "UTF-8 UNICODE (WITH BOM)" );
54      }
55  
56      @Test
57      public void detectUTF_16LE()
58      {
59          assertEncoding( new byte[]{(byte) 0xFF, (byte) 0xFE, (byte) 0xD6, (byte) 0x00, (byte) 0x41, (byte) 0x00}, StandardCharsets.UTF_16LE.name(), "LITTLE-ENDIAN UTF-16 UNICODE" );
60      }
61  
62      @Test
63      public void detectUTF_16BE()
64      {
65          assertEncoding( new byte[]{(byte) 0xFE, (byte) 0xFF, (byte) 0x00, (byte) 0xD6, (byte) 0x00, (byte) 0x41}, StandardCharsets.UTF_16BE.name(), "BIG-ENDIAN UTF-16 UNICODE" );
66      }
67  
68      @Test
69      public void detectExtendedAscii()
70      {
71          assertEncoding( new byte[]{'a', 'b', 'c', 'A', 'B', 'C', '1', '2', '3', (byte) 0x96}, "UNKNOWN-8BIT", "NON-ISO EXTENDED-ASCII" );
72      }
73  
74      private void assertEncoding( byte[] data, String codeMime, String code )
75      {
76          rule.guessFileEncoding( data );
77          assertEquals( codeMime, rule.getCodeMime().toUpperCase() );
78          assertEquals( code, rule.getCode().toUpperCase() );
79      }
80  }
