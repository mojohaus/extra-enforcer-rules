32      /**
33       * Try to determine whether text is in some character code we can
34       * identify. It also identifies EBCDIC by converting it to ISO-8859-1.
35       *
36       * @return true if it could guess an encoding.
37       */
38      public boolean guessFileEncoding( byte[] buf )
39      {
40          int nbytes = buf.length;
41          int ucs_type;
42  
43          if ( looks_ascii( buf, nbytes ) )
44          {
45              if ( looks_utf7( buf, nbytes ) )
46              {
47                  code = "UTF-7 Unicode";
48                  code_mime = "utf-7";
49              }
50              else
51              {
52                  code = "ASCII";
53                  code_mime = "us-ascii";
54              }
55          }
56          else if ( looks_utf8_with_BOM( buf, nbytes ) )
57          {
58              code = "UTF-8 Unicode (with BOM)";
59              code_mime = "utf-8";
60          }
61          else if ( looks_utf8( buf, nbytes ) > 1 )
62          {
63              code = "UTF-8 Unicode";
64              code_mime = "utf-8";
65          }
66          else if ( ( ucs_type = looks_ucs16( buf, nbytes ) ) != 0 )
67          {
68              if ( ucs_type == 1 )
69              {
70                  code = "Little-endian UTF-16 Unicode";
71                  code_mime = "utf-16le";
72              }
73              else
74              {
75                  code = "Big-endian UTF-16 Unicode";
76                  code_mime = "utf-16be";
77              }
78          }
79          else if ( looks_latin1( buf, nbytes ) )
80          {
81              code = "ISO-8859";
82              code_mime = "iso-8859-1";
83          }
84          else if ( looks_extended( buf, nbytes ) )
85          {
86              code = "Non-ISO extended-ASCII";
87              code_mime = "unknown-8bit";
88          }
89          else
90          {
91              byte[] nbuf = from_ebcdic( buf, nbytes );
92  
93              if ( looks_ascii( nbuf, nbytes ) )
94              {
95                  code = "EBCDIC";
96                  code_mime = "ebcdic";
97              }
98              else if ( looks_latin1( nbuf, nbytes ) )
99              {
100                 code = "International EBCDIC";
101                 code_mime = "ebcdic";
102             }
103             else
104             { /* Doesn't look like text at all */
105                 type = "binary";
106                 return false;
107             }
108         }
109 
110         return true;
111     }

113     /*
114      * This table reflects a particular philosophy about what constitutes
115      * "text," and there is room for disagreement about it.
116      *
117      * Version 3.31 of the file command considered a file to be ASCII if
118      * each of its characters was approved by either the isascii() or
119      * isalpha() function.  On most systems, this would mean that any
120      * file consisting only of characters in the range 0x00 ... 0x7F
121      * would be called ASCII text, but many systems might reasonably
122      * consider some characters outside this range to be alphabetic,
123      * so the file command would call such characters ASCII.  It might
124      * have been more accurate to call this "considered textual on the
125      * local system" than "ASCII."
126      *
127      * It considered a file to be "International language text" if each
128      * of its characters was either an ASCII printing character (according
129      * to the real ASCII standard, not the above test), a character in
130      * the range 0x80 ... 0xFF, or one of the following control characters:
131      * backspace, tab, line feed, vertical tab, form feed, carriage return,
132      * escape.  No attempt was made to determine the language in which files
133      * of this type were written.
134      *
135      *
136      * The table below considers a file to be ASCII if all of its characters
137      * are either ASCII printing characters (again, according to the X3.4
138      * standard, not isascii()) or any of the following controls: bell,
139      * backspace, tab, line feed, form feed, carriage return, esc, nextline.
140      *
141      * I include bell because some programs (particularly shell scripts)
142      * use it literally, even though it is rare in normal text.  I exclude
143      * vertical tab because it never seems to be used in real text.  I also
144      * include, with hesitation, the X3.64/ECMA-43 control nextline (0x85),
145      * because that's what the dd EBCDIC->ASCII table maps the EBCDIC newline
146      * character to.  It might be more appropriate to include it in the 8859
147      * set instead of the ASCII set, but it's got to be included in *something*
148      * we recognize or EBCDIC files aren't going to be considered textual.
149      * Some old Unix source files use SO/SI (^N/^O) to shift between Greek
150      * and Latin characters, so these should possibly be allowed.  But they
151      * make a real mess on VT100-style displays if they're not paired properly,
152      * so we are probably better off not calling them text.
153      *
154      * A file is considered to be ISO-8859 text if its characters are all
155      * either ASCII, according to the above definition, or printing characters
156      * from the ISO-8859 8-bit extension, characters 0xA0 ... 0xFF.
157      *
158      * Finally, a file is considered to be international text from some other
159      * character code if its characters are all either ISO-8859 (according to
160      * the above definition) or characters in the range 0x80 ... 0x9F, which
161      * ISO-8859 considers to be control characters but the IBM PC and Macintosh
162      * consider to be printing characters.
163      */
164     private static final byte F = 0; /* character never appears in text */
165     private static final byte T = 1; /* character appears in plain ASCII text */
166     private static final byte I = 2; /* character appears in ISO-8859 text */
167     private static final byte X = 3; /* character appears in non-ISO extended ASCII (Mac, IBM PC) */

191     private boolean looks_ascii( byte[] buf, int nbytes )
192     {
193         for ( int i = 0; i < nbytes; i++ )
194         {
195             if ( text_chars[ unsignedByte( buf[ i ] ) ] != T )
196             {
197                 return false;
198             }
199         }
200         return true;
201     }
202 
203     private boolean looks_latin1( byte[] buf, int nbytes )
204     {
205         for ( int i = 0; i < nbytes; i++ )
206         {
207             int t = text_chars[ unsignedByte( buf[ i ] ) ];
208 
209             if ( t != T && t != I )
210             {
211                 return false;
212             }
213         }
214 
215         return true;
216     }

218     private boolean looks_extended( byte[] buf, int nbytes )
219     {
220         for ( int i = 0; i < nbytes; i++ )
221         {
222             int t = text_chars[ unsignedByte( buf[ i ] ) ];
223             if ( t != T && t != I && t != X )
224             {
225                 return false;
226             }
227         }
228         return true;
229     }

231     /*
232      * Decide whether some text looks like UTF-8. Returns:
233      *
234      *     -1: invalid UTF-8
235      *      0: uses odd control characters, so doesn't look like text
236      *      1: 7-bit text
237      *      2: definitely UTF-8 text (valid high-bit set bytes)
238      */
239     protected int looks_utf8( byte[] buf, int nbytes )
240     {
241         boolean gotone = false;
242 
243         for ( int i = 0; i < nbytes; i++ )
244         {
245             if ( ( unsignedByte( buf[ i ] ) & 0x80 ) == 0 )
246             { /* 0xxxxxxx is plain ASCII */
247                 /*
248                 * Even if the whole file is valid UTF-8 sequences,
249                 * still reject it if it uses weird control characters.
250                 */
251 
252                 if ( text_chars[ unsignedByte( buf[ i ] ) ] != T )
253                 {
254                     return 0;
255                 }
256             }
257             else
258             {
259                 if ( ( unsignedByte( buf[ i ] ) & 0x40 ) == 0 )
260                 { /* 10xxxxxx never 1st byte */
261                     return -1;
262                 }
263                 else
264                 {   /* 11xxxxxx begins UTF-8 */
265                     int following;
266 
267                     if ( ( unsignedByte( buf[ i ] ) & 0x20 ) == 0 )
268                     { /* 110xxxxx */
269                         following = 1;
270                     }
271                     else if ( ( unsignedByte( buf[ i ] ) & 0x10 ) == 0 )
272                     { /* 1110xxxx */
273                         following = 2;
274                     }
275                     else if ( ( unsignedByte( buf[ i ] ) & 0x08 ) == 0 )
276                     { /* 11110xxx */
277                         following = 3;
278                     }
279                     else if ( ( unsignedByte( buf[ i ] ) & 0x04 ) == 0 )
280                     { /* 111110xx */
281                         following = 4;
282                     }
283                     else if ( ( unsignedByte( buf[ i ] ) & 0x02 ) == 0 )
284                     { /* 1111110x */
285                         following = 5;
286                     }
287                     else
288                     {
289                         return -1;
290                     }
291 
292                     for ( int n = 0; n < following; n++ )
293                     {
294                         i++;
295                         if ( i >= nbytes )
296                         {
297                             return gotone ? 2 : 1;
298                         }
299 
300                         if ( ( unsignedByte( buf[ i ] ) & 0x80 ) == 0 || ( unsignedByte( buf[ i ] ) & 0x40 ) > 0 )
301                         {
302                             return -1;
303                         }
304                     }
305                     gotone = true;
306                 }
307             }
308         }
309 
310         return gotone ? 2 : 1;
311     }

313     /*
314      * Decide whether some text looks like UTF-8 with BOM. If there is no
315      * BOM, return -1; otherwise return the result of looks_utf8 on the
316      * rest of the text.
317      */
318     private boolean looks_utf8_with_BOM( byte[] buf, int nbytes )
319     {
320         if ( nbytes > 3 && unsignedByte( buf[ 0 ] ) == 0xef && unsignedByte( buf[ 1 ] ) == 0xbb && unsignedByte( buf[ 2 ] ) == 0xbf )
321         {
322             return looks_utf8( Arrays.copyOfRange( buf, 3, nbytes ), nbytes - 3 ) > 0;
323         }
324         return false;
325     }

327     private boolean looks_utf7( byte[] buf, int nbytes )
328     {
329         if ( nbytes > 4 && buf[ 0 ] == '+' && buf[ 1 ] == '/' && buf[ 2 ] == 'v' )
330         {
331             switch ( buf[ 3 ] )
332             {
333                 case '8':
334                 case '9':
335                 case '+':
336                 case '/':
337                     return true;
338                 default:
339                     return false;
340             }
341         }
342         return false;
343     }

345     private int looks_ucs16( byte buf[], int nbytes )
346     {
347         int bigend;
348         int i;
349 
350         char ubuf[] = new char[ nbytes ];
351 
352         if ( nbytes < 2 )
353         {
354             return 0;
355         }
356 
357         if ( unsignedByte( buf[ 0 ] ) == 0xff && unsignedByte( buf[ 1 ] ) == 0xfe )
358         {
359             bigend = 0;
360         }
361         else
362         {
363             if ( unsignedByte( buf[ 0 ] ) == 0xfe && unsignedByte( buf[ 1 ] ) == 0xff )
364             {
365                 bigend = 1;
366             }
367             else
368             {
369                 return 0;
370             }
371         }
372 
373         int ulen = 0;
374 
375         for ( i = 2; i + 1 < nbytes; i += 2 )
376         {
377             if ( bigend == 1 )
378             {
379                 ubuf[ ( ulen )++ ] = (char) ( unsignedByte( buf[ i + 1 ] ) + 256 * buf[ i ] );
380             }
381             else
382             {
383                 ubuf[ ( ulen )++ ] = (char) ( unsignedByte( buf[ i ] ) + 256 * buf[ i + 1 ] );
384             }
385 
386             if ( ubuf[ ulen - 1 ] == 0xfffe )
387             {
388                 return 0;
389             }
390             if ( ubuf[ ulen - 1 ] < 128 && text_chars[ (int) ubuf[ ulen - 1 ] ] != T )
391             {
392                 return 0;
393             }
394         }
395 
396         return 1 + bigend;
397     }

400      * This table maps each EBCDIC character to an (8-bit extended) ASCII
401      * character, as specified in the rationale for the dd(1) command in
402      * draft 11.2 (September, 1991) of the POSIX P1003.2 standard.
403      *
404      * Unfortunately it does not seem to correspond exactly to any of the
405      * five variants of EBCDIC documented in IBM's _Enterprise Systems
406      * Architecture/390: Principles of Operation_, SA22-7201-06, Seventh
407      * Edition, July, 1999, pp. I-1 - I-4.
408      *
409      * Fortunately, though, all versions of EBCDIC, including this one, agree
410      * on most of the printing characters that also appear in (7-bit) ASCII.
411      * Of these, only '|', '!', '~', '^', '[', and ']' are in question at all.
412      *
413      * Fortunately too, there is general agreement that codes 0x00 through
414      * 0x3F represent control characters, 0x41 a nonbreaking space, and the
415      * remainder printing characters.

418      * between old-style and internationalized examples of text.

471     /*
472      * Copy buf[0 ... nbytes-1] into out[], translating EBCDIC to ASCII.
473      */
474     private byte[] from_ebcdic( byte[] buf, int nbytes )
475     {
476         byte[] out = new byte[ nbytes ];
477         for ( int i = 0; i < nbytes; i++ )
478         {
479             out[ i ] = (byte) ebcdic_to_ascii[ unsignedByte( buf[ i ] ) ];
480         }
481         return out;
482     }

484     private int unsignedByte( byte value )
485     {
486         return value & 0xFF;
487     }
