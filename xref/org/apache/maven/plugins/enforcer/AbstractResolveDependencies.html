87      private Set<Artifact> getDependenciesToCheck( MavenProject project, ArtifactRepository localRepository )
88      {
89          Set<Artifact> dependencies = null;
90          try
91          {
92              DependencyNode node = treeBuilder.buildDependencyTree( project, localRepository, null );
93              
94              if( isSearchTransitive() )
95              {
96                  dependencies  = getAllDescendants( node );
97              }
98              else if ( node.getChildren() != null )
99              {
100                 dependencies = new HashSet<Artifact>();
101                 for( DependencyNode depNode : node.getChildren() )
102                 {
103                     dependencies.add( depNode.getArtifact() );
104                 }
105             }
106         }
107         catch ( DependencyTreeBuilderException e )
108         {
109             // otherwise we need to change the signature of this protected method
110             throw new RuntimeException( e );
111         }
112         return dependencies;
113     }

115     private Set<Artifact> getAllDescendants( DependencyNode node )
116     {
117         Set<Artifact> children = null; 
118         if( node.getChildren() != null )
119         {
120             children = new HashSet<Artifact>();
121             for( DependencyNode depNode : node.getChildren() )
122             {
123                 try
124                 {
125                     Artifact artifact = depNode.getArtifact();
126     
127                     resolver.resolve( artifact, remoteRepositories, localRepository );
128                     
129                     children.add( artifact );
130     
131                     Set<Artifact> subNodes = getAllDescendants( depNode );
132                     
133                     if( subNodes != null )
134                     {
135                         children.addAll(subNodes);
136                     }
137                 }
138                 catch ( ArtifactResolutionException e )
139                 {
140                     getLog().warn( e.getMessage() );
141                 }
142                 catch ( ArtifactNotFoundException e )
143                 {
144                     getLog().warn( e.getMessage() );
145                 }
146             }
147         }
148         return children;
149     }

181     /**
182      * Convert a wildcard into a regex.
183      *
184      * @param wildcard the wildcard to convert.
185      * @return the equivalent regex.
186      */
187     protected static String asRegex(String wildcard)
188     {
189         StringBuilder result = new StringBuilder( wildcard.length() );
190         result.append( '^' );
191         for ( int index = 0; index < wildcard.length(); index++ )
192         {
193             char character = wildcard.charAt( index );
194             switch ( character )
195             {
196                 case '*':
197                     result.append( ".*" );
198                     break;
199                 case '?':
200                     result.append( "." );
201                     break;
202                 case '$':
203                 case '(':
204                 case ')':
205                 case '.':
206                 case '[':
207                 case '\\':
208                 case ']':
209                 case '^':
210                 case '{':
211                 case '|':
212                 case '}':
213                     result.append( "\\" );
214                 default:
215                     result.append( character );
216                     break;
217             }
218         }
219         result.append( "(\\.class)?" );
220         result.append( '$' );
221         return result.toString();
222     }

231         public Pattern classifier;

233         public List<Pattern> ignores = new ArrayList<Pattern>();

235         public IgnorableDependency applyIgnoreClasses( String[] ignores, boolean indent )
236         {
237             String prefix = indent ? "  " : "";
238             for ( String ignore : ignores )
239             {
240                 getLog().info( prefix + "Adding ignore: " + ignore );
241                 ignore = ignore.replace( '.', '/' );
242                 String pattern = asRegex( ignore );
243                 getLog().debug( prefix + "Ignore: " + ignore + " maps to regex " + pattern );
244                 this.ignores.add( Pattern.compile( pattern ) );
245             }
246             return this;
247         }
248 
249         public boolean matchesArtifact( Artifact dup )
250         {
251             return ( artifactId == null || artifactId.matcher( dup.getArtifactId() ).matches() )
252                 && ( groupId == null || groupId.matcher( dup.getGroupId() ).matches() )
253                 && ( classifier == null || classifier.matcher( dup.getClassifier() ).matches() )
254                 && ( type == null || type.matcher( dup.getType() ).matches() );
255         }
256 
257         public boolean matches(String className)
258         {
259             for ( Pattern p : ignores )
260             {
261                 if ( p.matcher( className ).matches() )
262                 {
263                     return true;
264                 }
265             }
266             return false;
267         }
