34  import java.util.regex.Pattern;

40  /**
41   * Bans duplicate classes on the classpath.
42   */
43  public class BanDuplicateClasses
44      extends AbstractResolveDependencies
45  {
46  
47      /**
48       * The failure message
49       */
50      private String message;
51      
52      /**
53       * List of classes to ignore. Wildcard at the end accepted
54       */
55      private String[] ignoreClasses;
56  
57      /**
58       * If {@code false} then the rule will fail at the first duplicate, if {@code true} then the rule will fail at
59       * the end.
60       */
61      private boolean findAllDuplicates;
62  
63      private List<Dependency> dependencies;
64      
65      /**
66       * Only verify dependencies with one of these scopes
67       */
68      private List<String> scopes;
69  
70      @Override
71      protected void handleArtifacts( Set<Artifact> artifacts ) throws EnforcerRuleException
72      {
73          List<IgnorableDependency> ignorableDependencies = new ArrayList<IgnorableDependency>();
74          if ( ignoreClasses != null )
75          {
76              IgnorableDependency ignorableDependency = new IgnorableDependency();
77              ignorableDependency.applyIgnoreClasses( ignoreClasses, false );
78              ignorableDependencies.add( ignorableDependency );
79          }
80          if ( dependencies != null )
81          {
82              for ( Dependency dependency : dependencies )
83              {
84                  getLog().info( "Adding ignorable dependency: " + dependency );
85                  IgnorableDependency ignorableDependency = new IgnorableDependency();
86                  if ( dependency.getGroupId() != null )
87                  {
88                      ignorableDependency.groupId = Pattern.compile( asRegex( dependency.getGroupId() ) );
89                  }
90                  if ( dependency.getArtifactId() != null )
91                  {
92                      ignorableDependency.artifactId = Pattern.compile( asRegex( dependency.getArtifactId() ) );
93                  }
94                  if ( dependency.getType() != null )
95                  {
96                      ignorableDependency.type = Pattern.compile( asRegex( dependency.getType() ) );
97                  }
98                  if ( dependency.getClassifier() != null )
99                  {
100                     ignorableDependency.classifier = Pattern.compile( asRegex( dependency.getClassifier() ) );
101                 }
102                 ignorableDependency.applyIgnoreClasses( dependency.getIgnoreClasses(), true );
103                 ignorableDependencies.add( ignorableDependency );
104             }
105         }
106         
107         Map<String, Artifact> classNames = new HashMap<String, Artifact>();
108         Map<String, Set<Artifact>> duplicates = new HashMap<String, Set<Artifact>>();
109         for ( Artifact o : artifacts )
110         {
111             if( scopes != null && !scopes.contains( o.getScope() ) )
112             {
113                 if( getLog().isDebugEnabled() )
114                 {
115                     getLog().debug( "Skipping " + o.toString() + " due to scope" );
116                 }
117                 continue;
118             }
119             File file = o.getFile();
120             getLog().debug( "Searching for duplicate classes in " + file );
121             if ( file == null || !file.exists() )
122             {
123                 getLog().warn( "Could not find " + o + " at " + file );
124             }
125             else if ( file.isDirectory() )
126             {
127                 try
128                 {
129                     for ( String name : (List<String>) FileUtils.getFileNames( file, null, null, false ) )
130                     {
131                         getLog().debug( "  " + name );
132                         checkAndAddName( o, name, classNames, duplicates, ignorableDependencies );
133                     }
134                 }
135                 catch ( IOException e )
136                 {
137                     throw new EnforcerRuleException(
138                         "Unable to process dependency " + o.toString() + " due to " + e.getLocalizedMessage(), e );
139                 }
140             }
141             else if ( file.isFile() && "jar".equals( o.getType() ) )
142             {
143                 try
144                 {
145                     //@todo use UnArchiver as defined per type
146                     JarFile jar = new JarFile( file );
147                     try
148                     {
149                         for ( JarEntry entry : Collections.<JarEntry>list( jar.entries() ) )
150                         {
151                             checkAndAddName( o, entry.getName(), classNames, duplicates, ignorableDependencies );
152                         }
153                     }
154                     finally
155                     {
156                         try
157                         {
158                             jar.close();
159                         }
160                         catch ( IOException e )
161                         {
162                             // ignore
163                         }
164                     }
165                 }
166                 catch ( IOException e )
167                 {
168                     throw new EnforcerRuleException(
169                         "Unable to process dependency " + o.toString() + " due to " + e.getLocalizedMessage(), e );
170                 }
171             }
172         }
173         if ( !duplicates.isEmpty() )
174         {
175             Map<Set<Artifact>, List<String>> inverted = new HashMap<Set<Artifact>, List<String>>();
176             for ( Map.Entry<String, Set<Artifact>> entry : duplicates.entrySet() )
177             {
178                 List<String> s = inverted.get( entry.getValue() );
179                 if ( s == null )
180                 {
181                     s = new ArrayList<String>();
182                 }
183                 s.add( entry.getKey() );
184                 inverted.put( entry.getValue(), s );
185             }
186             StringBuilder buf = new StringBuilder( message == null ? "Duplicate classes found:" : message );
187             buf.append( '\n' );
188             for ( Map.Entry<Set<Artifact>, List<String>> entry : inverted.entrySet() )
189             {
190                 buf.append( "\n  Found in:" );
191                 for ( Artifact a : entry.getKey() )
192                 {
193                     buf.append( "\n    " );
194                     buf.append( a );
195                 }
196                 buf.append( "\n  Duplicate classes:" );
197                 for ( String className : entry.getValue() )
198                 {
199                     buf.append( "\n    " );
200                     buf.append( className );
201                 }
202                 buf.append( '\n' );
203             }
204             throw new EnforcerRuleException( buf.toString() );
205         }
206 
207     }

209     private void checkAndAddName( Artifact artifact, String name, Map<String, Artifact> classNames,
210                                   Map<String, Set<Artifact>> duplicates, Collection<IgnorableDependency> ignores )
211         throws EnforcerRuleException
212     {
213         if ( !name.endsWith( ".class" ) )
214         {
215             return;
216         }
217         
218         for ( IgnorableDependency c : ignores )
219         {
220             if ( c.matchesArtifact( artifact ) && c.matches( name ) )
221             {
222                 if( classNames.containsKey( name ) )
223                 {
224                     getLog().debug( "Ignoring excluded class " + name );
225                 }
226                 return;
227             }
228         }
229 
230         if ( classNames.containsKey( name ) )
231         {
232             Artifact dup = classNames.put( name, artifact );
233             if ( !( findAllDuplicates && duplicates.containsKey( name ) ) )
234             {
235                 for ( IgnorableDependency c : ignores )
236                 {
237                     if ( c.matchesArtifact( artifact ) && c.matches(name) )
238                     {
239                         getLog().debug( "Ignoring duplicate class " + name );
240                         return;
241                     }
242                 }
243             }
244             
245             if ( findAllDuplicates )
246             {
247                 Set<Artifact> dups = duplicates.get( name );
248                 if ( dups == null )
249                 {
250                     dups = new LinkedHashSet<Artifact>();
251                     dups.add( dup );
252                 }
253                 dups.add( artifact );
254                 duplicates.put( name, dups );
255             }
256             else
257             {
258                 StringBuilder buf = new StringBuilder( message == null ? "Duplicate class found:" : message );
259                 buf.append( '\n' );
260                 buf.append( "\n  Found in:" );
261                 buf.append( "\n    " );
262                 buf.append( dup );
263                 buf.append( "\n    " );
264                 buf.append( artifact );
265                 buf.append( "\n  Duplicate classes:" );
266                 buf.append( "\n    " );
267                 buf.append( name );
268                 buf.append( '\n' );
269                 buf.append( "There may be others but <findAllDuplicates> was set to false, so failing fast" );
270                 throw new EnforcerRuleException( buf.toString() );
271             }
272         }
273         else 
274         {
275             classNames.put( name, artifact );
276         }
277     }
